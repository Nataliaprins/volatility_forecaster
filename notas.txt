función 2: calcula residuales(pronóstico,serie=70 ) -> residuales

        pronostico = calcula pronostico(modelo,serie)
        residuales = calcula residuales(pronostico,serie)

función 4: calcula errores en test(serie=100, modelo_mejorado) -> errores en test
     pronostico = calcula pronostico(modelo,serie=100)
     toma_los ultimos_30 = serie[70:]

     calcula errores(y_true,y_pred)-> metricas

función 5: pronostico n pasos hacia adelante (modelo, serie=100, horizonte=n)
    for i in n:
        pronostico = calcula pronostico(modelo,serie)
        serie = serie + pronostico
    return serie

    for i in n:
